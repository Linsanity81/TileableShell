75  typename Mesh::Scalar sqr_length = (_ci.p0 - _ci.p1).sqrnorm();

82 template<class MeshT>
83 void ModEdgeLengthT<MeshT>::set_error_tolerance_factor(double _factor) {
84  if (_factor >= 0.0 && _factor <= 1.0) {
85  // the smaller the factor, the smaller edge_length_ gets
86  // thus creating a stricter constraint
87  // division by error_tolerance_factor_ is for normalization
88  typename Mesh::Scalar edge_length = edge_length_ * static_cast<typename Mesh::Scalar>(_factor / this->error_tolerance_factor_);
89  set_edge_length(edge_length);
90  this->error_tolerance_factor_ = _factor;
91  }
92 }
